{Pumpkin Stuff}

    {% pictures
        small=pic/glass,     % small picture
        big=pic/ingredients  % big picture
    }
    
    \introduction{
        
    }
    
    \ingredients{%
        1 box & Vanilla/White/Yellow Cake Mix \\
        1/2 cup & butter\\
        1 & egg\\
        Middle: & \\
        1 small can & Pumpkin\\
        2/3 cup & Evaporated Milk\\
        3/4 cup & Sugar\\
        2 & eggs\\
        1/2 tsp & Vanilla\\
        2 tsp & Cinnamon\\
        Top: & \\
        1 remaining cup & Cake Mix\\
        1 tsp & Cinnamon\\
        1/4 cup & Sugar\\
        1/4 cup & Butter\\
    }
    
    \preparation{%
        \step Preheat the oven, and grease and flour the baking pan.
        \step Reserve 1 cup of the cake mix for the top layer.  Combine the 1/2 cup butter and one egg with all but the one cup of cake mix.  Press into bottom of a 9x13 pan.
        \step Mix all ingredients for the middle pumpkin layer and spread over the bottom crust.
        \step Use the reserved cup of cake mix and mix with cinnamon 1/4 cup of sugar and cut 1/4 cup of butter into it and sprinkle over top of the pumpkin.
        \step Bake for 45 minutes.
    }
