{Lemon Cheese Braid}
    
    \graph
    {% pictures
        small=pic/glass,     % small picture
        big=pic/ingredients  % big picture
    }
    
    %\introduction{

    %}
    
    \ingredients{%
		1 pkg & Yeast \\
        3 Tbls & Warm Water \\
        1/4 cup & Sugar \\
        1/3 cup & Warm Milk \\
        1/4 cup & Butter, melted \\
        2 & Eggs \\
        1/2 tsp & Salt \\
        3 - 3 1/2 cups & Flour \\
        Filling: \\
        2 pkgs (18 oz) & Cream Cheese, softened \\
        1/2 cup & Sugar \\
        1 & Egg \\
        1 tsp & Grated Lemon Peel \\
        Icing: \\
        1/2 cup & Powdered Sugar \\
        2 - 3 tsp & Milk \\
        1/4 tsp & Vanilla \\
    }
    
    \preparation{%
        \step Proof yest in the warm water for 5 minutes.  Add sugar, eggs, butter salt and 2 cups of flour.  Blend for 2 minutes.  Stir in remaining flour.
        \step Knead for 6-8 minutes.  Let rise for 1 hour.
        \step Beat filling until fluffy.
        \step Punch down dough and roll into a 12"x14" rectangle.  Spread filling onto center third of dough.  Cut symmetrical strips on the outer thirds and braid.
        \step Let raise again and bake.
    }
