\begin{recipe}
[% 
    preparationtime = {\unit[2 1/2]{h}},
    bakingtime={\unit[25-30]{min}},
    bakingtemperature={\protect\bakingtemperature{
        fanoven=\unit[425]{°F}}},
        %topbottomheat=\unit[195]{°C},
        %topheat=\unit[195]{°C},
        %gasstove=Level 2}},
    %portion = {\portion{16}},
    %calory={\unit[3]{kJ}},
    source = {Mrs. Drake}
]
{White Bread}
    
    \graph
    {% pictures
        small=pic/MrsDrakeBread,     % small picture
        big=pic/ingredients  % big picture
    }
    
    %\introduction{

    %}
    
    \ingredients{%
        1/2 cup & Warm Water \\
        2 pkg & Yeast \\
        1 3/4 cups & Lukewarm Liquid: \\
        3/4 cup & Scalded Milk + \\
        1 cup & Water to cool \\
        7 cups & Flour \\
        3 Tbls & Sugar \\
        1 Tbls & Salt \\
        2 Tbls & Shortening \\
    }
    
    \preparation{%
        \step Proof the yeast in the 1/2 cup water.  Add liquids.  Stir in half the flour and remaining ingredients.  Beat until smooth.  Use hands to mix in enough flour for dough to clean bowl.
        \step Turn onto floured board.  Cover and let rest for 10 - 15 minutes.
        \step Knead for 10 minutes.
        \step Cover and let rest for 1 hour.
        \step Punch down and lest rest for a half hour.
        \step Divide into 2 parts and let rest for 10 minutes.
        \step Shape into loaves and let raise.
        \step Bake until golden.
    }
    
    %\suggestion[Finishing]
    %{%
        
    %}
    
    \suggestion{%
		Don't skimp on the kneading time.  The bread needs it.
    }
    
    \hint{%
        For a less yeasty tasting bread reduce the amount of yeast to 1 3/4 Tablespoons and increase rising times.  And for overly fluffy bread reduce yeast to 1 1/5 Tablespoons.
    }
    
\end{recipe}
