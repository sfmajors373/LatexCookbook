{Hamburger Buns}
    
    \graph
    {% pictures
        small=pic/glass,     % small picture

    }
    
    %\introduction{

    %}
    
    \ingredients{%
        3/4 - 1 cup & Lukewarm Water \\
        2 Tbls & Butter, room temp \\
        1 & Egg \\
        3 1/2 cups & Flour \\
        1/4 cup & Sugar \\
        1 1/4 tsp & Salt \\
        1 Tbls & Yeast \\
        Topping: \\
        3 Tbls & Melted Butter \\
    }
    
    \preparation{%
        \step Mix and knead all dough ingredients.  Cover and let raise for 1 - 2 hours.
        \step Gently deflate and divide into 8 pieces.  Shape into a round ball and flatten to 3" diameter.  Place on baking sheet and let raise for another hour.
        \step Brush the buns with half the melted butter.  Bake until golden and brush with remaining butter.
    }
