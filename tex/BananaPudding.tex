    %\introduction{

    \ingredients{%
        3 cups & Cold Milk\\
        2 packages & Instant Vanilla Pudding\\
        1 box & Nilla Wafers\\
        3 & Bananas, sliced\\
        1 tub & Cool Whip\\
    }
    
    \preparation{%
        \step Mix milk and pudding mix for two minutes.
        \step Let stand for 5 minutes.
        \step Arrange half the Nilla Wafers on bottom of 2 quart serving bowl.
        \step Add a layer of bananas over the cookies.
        \step Carefully pour a layer of pudding over the bananas, and repeat the layering process.
        \step Spread Cool Whip on top.  
        \step Chill for 3 hours.
    }
