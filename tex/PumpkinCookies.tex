{Pumpkin Cookies}

    {% pictures
        small=pic/glass,     % small picture

    }
    
    %\introduction{

    %}
    
    \ingredients{%
        2 1/2 cups & Flour \\
        1 tsp & Baking Soda\\
        1 tsp & Baking Powder\\
        1 tsp & Cinnamon\\
        1/2 tsp & Nutmeg\\
        1 1/2 cups & Sugar\\
        1/2 cup & Butter\\
        1 cup & Pumpkin\\
        1 & Egg\\
        1 tsp & Vanilla\\
        Glaze:\\
        2 cups & Powdered Sugar\\
        3 Tbsp & Milk\\
        1 Tbsp & Butter\\
        1 tsp & Vanilla\\
    }
    
    \preparation{%
        \step Beat butter and sugar.
        \step Add wet ingredients.  Mix in dry ingredients.
        \step Drop onto cookie sheet and bake for 15 - 18 minutes.
        \step Mix together powdered sugar, milk, 1 Tbsp butter and 1 tsp vanilla in separate bowl to make glaze.
        \step Drizzle glaze over cooled cookies.

    }
