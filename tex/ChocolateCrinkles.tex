{Chocolate Crinkles}

        small=pic/glass,     % small picture

    }
    
    %\introduction{

    %}
    
    \ingredients{%
        3/4 cup & Butter \\
	1/2 cup & Cocoa Powder \\
	1 cup & Sugar \\
	2 & Eggs \\
	2 tsp & Vanilla \\
	2 cups & Flour \\
	1 tsp & Baking Powder \\
	1 tsp & Baking Soda \\
	1 16oz bag & Chocolate Chips \\
	3/4 cup & Powdered Sugar \\
    }
    
    \preparation{%
        \step Combine butter, cocoa and sugar.
        \step Beat in eggs and vanilla with electric mixer.
        \step Mix in flour, bakins soda, and baking powder by hand.  Stir in chocolate chips.
	\step Form balls and roll in powdered sugar.
	\step Bake for 12 minutes.
    }
