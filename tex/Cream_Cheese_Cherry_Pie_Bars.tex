{Cream Cheese Cherry Pie Bars}
    
    \graph

        small=pic/CherryBars,     % small picture
        big=pic/ingredients  % big picture
    }
    
    \introduction{
        Delicious, and yummy with a homemade cherry filling
    }
    
    \ingredients{%
        3 cups & Flour \\
        1 cup & Sugar\\
        1 tsp & Baking Powder\\
        1 cup & Cold Butter\\
        1 egg\\
        Filling:&\\
        24 (or 16) oz & Cream Cheese\\
        1 cup & Sugar\\
        2 tsp & Vanilla\\
        2 cans & Cherry Pie Filling\\
    }
    
    \preparation{%
        \step Preheat the oven, and lightly grease a 9x13 baking pan.
        \step Mix crumb and press half into bottom of pan.  Bake for 8-10 minutes.
        \step Mix cream cheese, sugar and vanilla and spread over the crust.  Layer the cherry filling over top.
        \step Sprinkle remaining crumb mixture over the filling.
        \step Bake 45-50 minutes.
    }
