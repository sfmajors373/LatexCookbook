{Ultimate Brownies}
    
    \graph
    {% pictures
        small=pic/glass,     % small picture
        big=pic/ingredients  % big picture
    }
    
    \introduction{
        Crunchy on the top and chewy on the inside.
    }
    
    \ingredients{%
        1 cup & Butter\\
        1 2/3 cup & Sifted Flour\\
        6 ounces & Unsweetened Baking Chocolate\\
        5 & Large Eggs\\
        2 tsp & Vanilla\\
        3 1/4 cup & sugar
    }
    
    \preparation{%
        \step Heat oven to 425F.  Grease and flour a 9x13 pan.
        \step Melt butter and chocolate in a double boiler.  Remove from heat and let cool.
        \step In mixer bowl with whisk attachment combine eggs, vanilla, and sugar; beat on high for ten minutes.  Reduce speed to low.  Add the chocolate mixture.  Continue beating to combine.  Add flour just to combine.
        \step Bake 10 minutes.  Reduce heat to 350 F and bake for an addition 15 minutes.
    }
