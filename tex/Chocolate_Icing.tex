    \introduction{
        This chocolate icing recipe uses cream cheese to balance the sweetness and the richness for a nice flavor that is not overwhelming.
    }
    
    \ingredients{%
        8 oz pkg & Cream Cheese\\
        1/3 cup & Butter\\
        6 cups & Powdered Sugar\\
        1/2 cup & Cocoa\\
        1/3 - 1/2 cup & milk\\
        1 3/4 tsp & vanilla
    }
    
    \preparation{%
        \step Cream butter and cream cheese.
        \step Alternately add milk and powdered sugar.
        \step Add vanilla.
    }
