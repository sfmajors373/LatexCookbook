{Raspberry Cream Cheese Bars}
    
    \graph
    {% pictures
        small=pic/glass,     % small picture
        big=pic/ingredients  % big picture
    }
    
    %\introduction{

    %}
    
    \ingredients{%
        Dough: 3/4 cup & Warm Water \\
        1 1/2 tsp & Yeast \\
        3 1/2 cups & Flour \\
        2 Tbls & Sugar \\
        1 & Egg \\
        1/2 cup & Butter, Softened \\
        1 Tbls & Milk or Sour Cream \\
        Raspberry Filling:  1/3 cup & Sugar \\
        1 Tbls + 1 tsp & Cornstarch \\
        1/3 cup & Water \\
        Cream Cheese:  1 pkg & Cream Cheese \\
        1 & Egg \\
        1/3 cup & Sugar \\
        1Tbls + 1tsp & Cornstarch \\
        1 tsp & Lemon Juice \\
        1 tsp & Vanilla \\
    }
    
    \preparation{%
        \step Proof yeast for 5 minutes.
        \step Mix in other dough ingredients.
        \step Knead for 8 - 10 minutes.  Allow to raise for 1 - 1 1/2 hours.
        \step Divide into 3 pieces.  Pat one into the bottom of a 9x13" pan.  Spread the raspberry filling over the bottom dough.  Pat the second portion of dough over the raspberry.  Spread the cream cheese mixture and then finally pat the final dough portion on top.
        \step Let rise for another 30 - 40 minutes.  Bake for 40 minutes.
    }
