    \ingredients{%
       Cookie: 3 1/2 cups & Flour \\
       1 1/2 cups & Cocoa Powder \\
       1 Tbls & Baking Soda \\
       1 tsp & Baking Powder \\
       1 cup & Butter \\
       2 cups & Sugar \\
       2 & Eggs \\
       2 cups & Buttermilk, room temp \\
       2 tsp & Vanilla \\
       Cream: 2 & Egg Whites \\
       1 1/2 cup & Sugar \\
       2 tsp & Corn Syrup \\
       1/4 tsp & Cream of Tartar \\
       1/2 tsp & Vanilla \\
    }
    
    \preparation{%
        \step Sift flour, cocoa, powder and soda.  Cream butter and sugar.  Add egg, buttermilk, and vanilla.  Mix in dry ingredients.
        \step Bake 1oz cookies for 12 minutes.  Allow to cool.
        \step Combine all cream ingredients except vanilla with 5 Tbls cold water.  Place over boiling water and mix for 7 minutes with electric mixer.  Remove from heat and add vanilla.
        \step Sandwich two cookies with a dollop of cream.
    }
