{Nut Filled Cookies}
    
    \graph
    {% pictures
        small=pic/glass,     % small picture
        big=pic/ingredients  % big picture
    }
    
    %\introduction{

    %}
    
    \ingredients{%
       6 1/2 cups & Flour \\
       1 lb & Butter\\
       1/2 pint & Cream\\
       10 & Egg Yolks\\
       1 box & Powdered Sugar
       Filling:\\
       10 & Egg Whites\\
       1 box & Powdered Sugar\\
       3 lbs & Ground Walnuts\\
    }
    
    \preparation{%
        \step Combine butter and egg yolks. Alternate additions of flour and cream.
        \step Roll into small balls.
        \step Refrigerate overnight; roll in powdered sugar.
        \step Beat egg whites until stiff.
        \step Add powdered sugar and beat well.
        \step Incorporate finely ground nuts with a spoon - do not beat.
        \step Bake for 10 minutes.

    }
    
    %\suggestion[Finishing]

    }
