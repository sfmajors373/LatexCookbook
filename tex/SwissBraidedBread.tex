    \introduction{
        Very good, but not exactly great for sandwiches.
    }
    
    \ingredients{%
        1 package & Dry Yeast\\
        1 1/4 cup & Milk\\
        3 1/2 cups & Flour\\
        1 1/2 tsp & Salt\\
        4 Tbls & Butter\\
        1 tsp & Sugar\\
    }
    
    \preparation{%
        \step Put 2 cups flour, yest, salt, and sugar in bowl.  Mix well.
        \step Soften butter and add to flour mixture.
        \step Heat milk to 110-120 F, add and stir until smooth.
        \step Add the rest of the flour and knead for 10 minutes.  
        \step Let rise.  Punch down.  Flatten into rectangle; cut into three strips and braid.
        \step Place on greased sheet and let raise for half hour.
        \step Bake for 40 minutes.
    }
