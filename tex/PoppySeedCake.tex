    \ingredients{%
        3 cups & Flour \\
        1 1/2 tsp & Baking Soda\\
        2 cups & Sugar\\
        8 oz & Oil\\
        4 & Eggs\\
        1 can & Evaporated Milk\\
        1 tsp & Vanilla\\
        1 jar & Poppyseed Filling\\
    }
    
    \preparation{%
        \step Preheat the oven, and grease and flour or spray the tube pan.
        \step Sift dry ingredients.
        \step Incorporate liquid ingredients and mix until smooth.
        \step Add poppyseed and mix on medium speed for two minutes.
        \step Bake in tube pan for 50 - 60 minutes.
    }
