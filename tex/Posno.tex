    \ingredients{%
        4 cups & Diced Potatoes \\
        1 cup & Celery \\
        1 cup & Carrots \\
        & Parsley \\
        1 cup & Onions \\
        1 clove & Garlic \\
        1 can & Diced Tomatoes \\
        2 containers & Fresh Mushrooms \\
        & Salt \\
        & Pepper \\
        8 quarts & Water \\
        1/2 cup & Oil \\
        4 Tbls & Flour \\
    }
    
    \preparation{%
        \step In pot, place potatoes, celery, carrots, parsley and tomatoes.
        \step In separate pan, brown onions, mushrooms, and garlic in 1/2 cup oil.  Add to soup.
        \step Make rue: Combine the cooked 1/2 cup oil and 4 Tbls flour.  Slightly brown then add 2 cups of broth.
        \step Add rue to soup.
    }
