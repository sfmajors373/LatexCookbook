    \ingredients{%
        2 1/2 cups & Water \\
        1 cup & Sugar\\
        2 heaping Tbsp & Cornstarch\\
        1 small box & Strawberry Jello\\
        2 cups & Crushed Strawberries\\
    }
    
    \preparation{%
        \step Boil water and sugar.  Add the cornstarch to thicken syrup.
        \step Mix in strawberry jello.  Let cool.
        \step Stir in crushed strawberries.
        \step Place in crust.
