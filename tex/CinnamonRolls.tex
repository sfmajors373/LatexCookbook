    \ingredients{%
        Bread: \\
        2 1/4 tsp & Yeast \\
        1 cup & Warm Milk \\
        1/2 cup & Sugar \\
        1/3 cup & Butter \\
        2 & Eggs \\
        4 cups & Flour \\
        Filling: \\
        1 cup & Brown Sugar \\
        2 1/2 tsp & Cinnamon \\
        1/3 cup & Butter \\
        Icing: \\
        1/2 cup & Butter \\
        1 1/2 cups & Powdered Sugar \\
        1/2 cup & Cream Cheese \\
        1/2 tsp & Vanilla \\        
    }
    
    \preparation{%
        \step Dissolve yeast in milk.  Mix in all other ingredients for the dough.
        \step Knead and let raise for 1 hour.
        \step Roll into 21"x16" rectangle.
        \step Spread filling over rectangle.  
        \step Roll the dough from the short end, and cut into 12 pieces.  Place into 9x13 stone.
        \step Bake for 10 minutes.
    }
