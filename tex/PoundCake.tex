[%

        fanoven=\unit[350]{°F}}},

{Pound Cake}
    
    \graph
    {% pictures
        small=pic/glass,     % small picture
        big=pic/ingredients  % big picture
    }
    
    \introduction{
        Great for breakfasts!
    }
    
    \ingredients{%
        3 cups & Flour \\
        3/4 tsp & Baking Powder\\
        1 cup & Butter, Softened\\
        1/2 cup & Shortening\\
        2 2/3 cup & Sugar\\
        5 & Eggs\\
        1 cup & Milk\\
        2 tsp & Vanilla\\
    }
    
    \preparation{%
        \step Do NOT preheat the oven.  Grease tube pan.
        \step Sift flour and baking powder.
        \step Beat butter, shortening, and sugar on low speed for 10 minutes.
        \step Add eggs one at a time, blending after each.  Alternately add flour and milk.  Add in vanilla.
        \step Bake for an hour and 20 minutes.  Turn off the oven and leave the cake in while the oven cools.
    }
    
    %\suggestion[Finishing]
    {%
        
    }
    
    \suggestion{%
        The vanilla can be substituted with vanilla butternut extract for extra goodness.
    }

  %  }
    
\end{recipe}
