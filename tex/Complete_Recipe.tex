    \ingredients{%
        2 cups & Flour \\
        2/3 cup & Cocoa\\
        1 tsp & Baking Powder\\
        1 tsp & Baking Soda\\
        3/4 cup & Butter\\
        1 3/4 cup & Sugar\\
        2 tsp & Vanilla\\
        2 & Eggs\\
        1 cup & Water
    }
    
    \preparation{%
        \step Preheat the oven, and grease and flour the baking pan.
        \step Beat butter, sugar, and vanilla in mixer until creamy.
        \step Add eggs and beat well.
        \step Alternate the addition of flour and water.
        \step Spread into pans.
        \step Bake 25 to 35 minutes or until an inserted toothpick comes out clean.
    }
