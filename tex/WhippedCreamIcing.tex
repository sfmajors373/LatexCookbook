    \ingredients{%
        2 Tbls & Cornstarch\\
        1 cup & Milk\\
        1 cup & Shortening\\
        1 cup & Sugar\\
        1 tsp & Vanilla\\
    }
    
    \preparation{%
        \step Heat cornstarch and milk in saucepan.  Stir continually until boiled.  (Careful, this happens very quickly)
        \step Remove mixture from heat and cool.  Lay saran wrap on the top of the mixture to prevent a skin from forming over it.
        \step  Beat remaining ingredients on high until fluffy.  This will take time.
        \step Beat in cooled milk mixture.
    }
