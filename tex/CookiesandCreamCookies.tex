    \ingredients{%
        3/4 cup & Butter \\
	1/2 cup & Brown Sugar \\
	2 & Eggs \\
	1 tsp & Vanilla \\
	1 pkg (4.2 oz) & Cookies and Cream Pudding Mix \\
	1 tsp & Baking Soda \\
	2 1/4 cup & Flour \\
	8 & Oreos \\
    }
    
    \preparation{%
        \step Beat softened butter and brown sugar.  Mix in eggs and vanilla.
        \step Stir in pudding mix, baking soda, and flour.
        \step Crunch the oreos and incorporate them into the dough.
        \step Drop by rounded teaspoon and bake for 8 - 12 minutes
    }
